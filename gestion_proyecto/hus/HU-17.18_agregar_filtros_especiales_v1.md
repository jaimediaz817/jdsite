### Filtro de Autor con Autocomplete (AJAX)
- **Tecnología**: `fetch()` vanilla JS (misma que usa `dashboard.js` para toggle/moderación)
- **Input**: campo de texto con clase `form-control`, dentro de `<fieldset class="filtro-fecha-creacion">` (consistente con blog_list.html)
- **Flujo AJAX**:
  1. Usuario escribe 2+ caracteres en el input
  2. JavaScript espera 300ms (debounce) para no saturar el backend
  3. `fetch('/blog/api/authors/autocomplete/?q=...')` hace petición GET
  4. Backend responde JSON: `{ "authors": [{"id":1, "username":"jaime", "display":"Jaime Díaz"}, ...] }`
  5. JavaScript pinta dropdown con sugerencias debajo del input
  6. Al hacer click en una sugerencia → se llena el input con el username → se envía el form
- **Dropdown**: `<div class="author-suggestions-dropdown">` con posicionamiento absolute
- **Cierre**: al hacer click fuera del input o del dropdown, se cierra
- **Backend endpoint**: busca en `User` (first_name, last_name, username) con `icontains`, máximo 8 resultados
- **Permisos**: superadmin ve todos los autores; autor normal solo ve su propio nombre
- **Parámetro GET**: `author=nombre` se conserva en paginación (ya lo hace `query_string`)
- **Limpiar**: el botón "Limpiar" resetea todos los parámetros incluyendo `author`

## Diseño de la solución

### Backend — Endpoint de Autocomplete (`views.py`)

```python
@require_http_methods(["GET"])
def api_authors_autocomplete(request):
    """Devuelve sugerencias de autores para el autocomplete del dashboard.
    
    Busca en User: first_name, last_name, username con icontains.
    Si el usuario es staff/superadmin ve todos; si no, solo su propio nombre.
    Retorna máximo 8 sugerencias.
    """
    q = request.GET.get("q", "").strip()
    if len(q) < 2:
        return JsonResponse({"authors": []})
    
    from django.contrib.auth.models import User
    
    if request.user.is_superuser or request.user.is_staff:
        users = User.objects.filter(
            Q(first_name__icontains=q) |
            Q(last_name__icontains=q) |
            Q(username__icontains=q)
        ).distinct()[:8]
    else:
        users = User.objects.filter(id=request.user.id)
    
    authors = []
    for u in users:
        display = u.get_full_name() or u.username
        authors.append({
            "id": u.id,
            "username": u.username,
            "display": display,
        })
    
    return JsonResponse({"authors": authors})
```

### JavaScript — Autocomplete AJAX + Mejoras Buscador (`dashboard.js`)

El archivo `dashboard.js` tendrá 2 bloques nuevos:

**Bloque 1: Autocomplete de autores (AJAX)**

```javascript
// =============================================
// Autocomplete AJAX de autores en el dashboard
// =============================================
// Usa fetch() vanilla JS (sin jQuery ni librerías externas).
// Patrón: debounce → fetch → dropdown → selección → submit form.

document.addEventListener('DOMContentLoaded', function() {
    const input = document.getElementById('author-autocomplete-input');
    const suggestionsDiv = document.getElementById('author-suggestions');
    if (!input || !suggestionsDiv) return;
    
    let debounceTimer;
    
    // 1. Escuchar input — debounce 300ms
    input.addEventListener('input', function() {
        clearTimeout(debounceTimer);
        const q = this.value.trim();
        
        if (q.length < 2) {
            suggestionsDiv.classList.add('d-none');
            return;
        }
        
        debounceTimer = setTimeout(() => {
            // 2. Petición AJAX via fetch()
            fetch('/blog/api/authors/autocomplete/?q=' + encodeURIComponent(q))
                .then(function(response) { return response.json(); })
                .then(function(data) {
                    if (data.authors.length === 0) {
                        suggestionsDiv.classList.add('d-none');
                        return;
                    }
                    
                    // 3. Pintar dropdown con sugerencias
                    suggestionsDiv.innerHTML = data.authors.map(function(a) {
                        return '<div class="author-suggestion-item" data-username="' + a.username + '">' + a.display + '</div>';
                    }).join('');
                    suggestionsDiv.classList.remove('d-none');
                    
                    // 4. Click en sugerencia → llenar input + submit form
                    suggestionsDiv.querySelectorAll('.author-suggestion-item').forEach(function(item) {
                        item.addEventListener('click', function() {
                            input.value = this.dataset.username;
                            suggestionsDiv.classList.add('d-none');
                            input.closest('form').submit();
                        });
                    });
                })
                .catch(function(error) {
                    console.error('Error autocomplete autor:', error);
                    suggestionsDiv.classList.add('d-none');
                });
        }, 300); // debounce 300ms
    });
    
// 5. Cerrar sugerencias al hacer click fuera
    document.addEventListener('click', function(e) {
        if (!input.contains(e.target) && !suggestionsDiv.contains(e.target)) {
            suggestionsDiv.classList.add('d-none');
        }
    });
});

// =============================================
// Bloque 2: Mejoras al buscador #blog-search-input
// =============================================
// - Botón X para limpiar (aparece/oculta según contenido)
// - Búsqueda en vivo con debounce (500ms)
// - Atajo Escape para limpiar y cerrar foco
// - Contador de resultados

(function() {
    var searchInput = document.getElementById('blog-search-input');
    if (!searchInput) return;

    var form = searchInput.closest('form');
    var debounceTimer;

    // 1. Botón X para limpiar — crear dinámicamente
    var clearBtn = document.createElement('button');
    clearBtn.type = 'button';
    clearBtn.className = 'btn btn-sm position-absolute';
    clearBtn.style.cssText = 'right: 60px; top: 50%; transform: translateY(-50%); z-index: 10; padding: 2px 6px; font-size: .75rem; color: #9ca3af; background: transparent; border: none; display: none;';
    clearBtn.innerHTML = '<i class="fas fa-times-circle"></i>';
    clearBtn.title = 'Limpiar búsqueda';
    clearBtn.setAttribute('aria-label', 'Limpiar búsqueda');

    // Insertar el botón X dentro del input-group
    var inputGroup = searchInput.closest('.input-group');
    if (inputGroup) {
        inputGroup.style.position = 'relative';
        inputGroup.appendChild(clearBtn);
    }

    // Mostrar/ocultar X según contenido
    function toggleClearBtn() {
        clearBtn.style.display = searchInput.value.trim() ? 'block' : 'none';
    }
    toggleClearBtn();
    searchInput.addEventListener('input', toggleClearBtn);

    // Click en X → limpiar input + submit
    clearBtn.addEventListener('click', function() {
        searchInput.value = '';
        toggleClearBtn();
        searchInput.focus();
        // Submit limpio (sin q)
        window.location.href = form.getAttribute('action') || window.location.pathname;
    });

    // 2. Búsqueda en vivo con debounce (500ms)
    searchInput.addEventListener('input', function() {
        clearTimeout(debounceTimer);
        var val = this.value.trim();
        debounceTimer = setTimeout(function() {
            if (val.length >= 2 || val.length === 0) {
                form.submit();
            }
        }, 500);
    });

    // 3. Atajo Escape → limpiar + cerrar foco
    searchInput.addEventListener('keydown', function(e) {
        if (e.key === 'Escape') {
            searchInput.value = '';
            toggleClearBtn();
            searchInput.blur();
        }
    });
})();
```

**Nota**: El debounce de 500ms para el buscador es independiente al debounce de 300ms del autocomplete de autor (son IDs diferentes: `#blog-search-input` vs `#author-autocomplete-input`).
